# original prior predictive samples
k = GPy.kern.RBF(input_dim=1,lengthscale=1)

X_dom = np.linspace(-2.,12.,1000) # 1000 points evenly spaced over [0,1]

X_dom = X_dom[:,None] # reshape X to make it n*D

mu = np.zeros((1000)) # vector of the means

Sigma = k.K(X_dom,X_dom) # covariance matrix

# Generate 20 sample path with mean mu and covariance C
Z = np.random.multivariate_normal(mu,Sigma,10)

plt.figure() # open new plotting window
for i in range(10):
    plt.plot(X_dom[:],Z[i,:])
plt.xlabel('X')
plt.ylabel('Y')
plt.show()
